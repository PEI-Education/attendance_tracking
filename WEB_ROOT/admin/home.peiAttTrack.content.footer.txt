~[if.~(studentscreenaccess;attendance.html)=1]
    <script>
        $j(function() {
            let requests = { ~[tlist_sql;
                select
                    count(case when concern_2 = 1 then 1 else null end) as letter1,
                    count(case when concern_3 = 1 then 1 else null end) as letter2,
                    count(case when concern_4 = 1 then 1 else null end) as letter3
                from u_pei_att_track
                where
                    substr(termid,0,2) = ~(curyearid)
                    
                order by 1;
                ]
                    letter1: ~(1),
                    letter2: ~(2),
                    letter3: ~(3)
                [/tlist_sql]
                }
            
            let sum = Object.values(requests).reduce((a, b) => a + b, 0);
            if (sum > 0) {     
                $j('h1').after(`<div id="attAlert" class="feedback-alert">Staff have requested ${sum} written attendance notifications. See the list of students on the <a href="/admin/pei_reports/attendance_tracking.html">Attendance Concerns Dashboard</a>.</div>`);
            }
        });
    </script>
